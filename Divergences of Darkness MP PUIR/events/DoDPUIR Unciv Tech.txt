#this bad boy handles giving the uncivs their specific techs that they don't have from their starting reforms from PUIR, the technologies given are based on DoD's Technology names, so if you try to use this event, make sure the technology names match up.

#Land Reform
country_event = {
	id = 79797980
	allow_multiple_instances = yes 
	title = "Land Reform Tech"
	desc = "And thus you have been given your tech."
	
	trigger = {
		civilized = no
		OR = {
			land_reform = yes_land_reform
			land_reform = land_reform_two
		}
		NOT = { has_country_flag = dodpuir_unciv_tech }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_country_flag = dodpuir_unciv_tech
	}
	
	option = {
		name = "Have fun."
		random_owned = {
			owner = {
				limit = {
					land_reform = land_reform_two
				}
				activate_technology = freedom_of_trade
				activate_technology = high_pressure_steam_engine 
				activate_technology = mechanized_mining
				activate_technology = piston_steam_engine_power
			}
		}
		random_owned = {
			owner = {
				limit = {
					land_reform = yes_land_reform
				}
				activate_technology = mechanized_mining
				activate_technology = piston_steam_engine_power
			}
		}
	}
	
}
